# 695. Max Area of Island
## STEP1
- 何も見ずに解いてみる
- 前の問題と比較して面積を答える必要があるので見つかった land を数えることとした。
```python
from collections import deque


class Solution:
    def maxAreaOfIsland(self, grid: List[List[int]]) -> int:
        WATER = 0  # otherwise (1): LAND
        num_rows = len(grid)
        num_columns = len(grid[0])
        checked = [[False for _ in range(num_columns)] for _ in range(num_rows)]

        def is_in_grid(row: int, column: int) -> bool:
            return 0 <= row < num_rows and 0 <= column < num_columns

        def traverse_and_count_lands(row: int, column: int) -> int:
            lands = deque([(row, column)])
            checked[row][column] = True
            area = 0
            while lands:
                row, column = lands.popleft()
                area += 1
                for (i, j) in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                    next_row = row + i
                    next_column = column + j
                    if not is_in_grid(next_row, next_column):
                        continue
                    if grid[next_row][next_column] == WATER:
                        continue
                    if checked[next_row][next_column]:
                        continue
                    lands.append((next_row, next_column))
                    checked[next_row][next_column] = True
            return area

        max_area = 0
        for row in range(num_rows):
            for column in range(num_columns):
                if grid[row][column] == WATER:
                    continue
                if checked[row][column]:
                    continue
                area = traverse_and_count_lands(row, column)
                max_area = max(max_area, area)
        return max_area
```
#### memo
n = len(grid), m = len(grid[0]) として  
時間計算量: $O(nm)$  
空間計算量: $O(min(n, m))$

from collections import deque


class Solution:
    def maxAreaOfIsland(self, grid: List[List[int]]) -> int:
        WATER = 0  # otherwise (1): LAND
        num_rows = len(grid)
        num_columns = len(grid[0])
        checked = [[False for _ in range(num_columns)] for _ in range(num_rows)]

        def should_visited(row: int, column: int) -> bool:
            if not (0 <= row < num_rows and 0 <= column < num_columns):
                return False
            if grid[row][column] == WATER:
                return False
            return not checked[row][column]

        def get_island_area(row: int, column: int) -> int:
            lands = deque([(row, column)])
            area = 0
            while lands:
                row, column = lands.popleft()
                if not should_visited(row, column):
                    continue
                area += 1
                checked[row][column] = True
                lands.append((row + 1, column))
                lands.append((row - 1, column))
                lands.append((row, column + 1))
                lands.append((row, column - 1))
            return area

        max_area = 0
        for row in range(num_rows):
            for column in range(num_columns):
                if grid[row][column] == WATER:
                    continue
                if checked[row][column]:
                    continue
                area = get_island_area(row, column)
                max_area = max(max_area, area)
        return max_area
