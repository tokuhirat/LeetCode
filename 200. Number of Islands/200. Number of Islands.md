# 200. Number of Islands
## STEP1
- 何も見ずに解いてみる
```python
class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        num_row = len(grid)
        num_column = len(grid[0])
        checked = [[False for _ in range(num_column)] for _ in range(num_row)]

        def is_in_grid(row: int, column: int) -> bool:
            return 0 <= row < num_row and 0 <= column < num_column

        def search_adjacent_lands(row: int, column: int):
            lands = [(row, column)]
            checked[row][column] = True
            while lands:
                (row, column) = lands.pop(0)
                for i, j in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                    next_row = row + i
                    next_column = column + j
                    if not is_in_grid(next_row, next_column):
                        continue
                    if grid[next_row][next_column] == "0":
                        continue
                    if checked[next_row][next_column]:
                        continue        
                    lands.append((next_row, next_column))
                    checked[next_row][next_column] = True

        num_island = 0
        for row in range(num_row):
            for column in range(num_column):
                if grid[row][column] == "0":
                    continue
                if checked[row][column]:
                    continue
                search_adjacent_lands(row, column)
                num_island += 1
        return num_island
```
