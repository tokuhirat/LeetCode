# 102. Binary Tree Level Order Traversal
## STEP1
- 何も見ずに解いてみる
- 各階層ごとなので BFS で書くのがわかりやすそう
```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def levelOrder(self, root: Optional[TreeNode]) -> List[List[int]]:
        if root is None:
            return []
        
        result = []
        nodes = [root]
        while nodes:
            values = []
            next_nodes = []
            for node in nodes:
                values.append(node.val)
                if node.left is not None:
                    next_nodes.append(node.left)
                if node.right is not None:
                    next_nodes.append(node.right)
            result.append(values)
            nodes = next_nodes
        return result
```
#### memo
時間計算量: $O(n)$  
空間計算量: $O(n)$

練習として DFS で書いてみる
```python
from collections import defaultdict


class Solution:
    def levelOrder(self, root: Optional[TreeNode]) -> List[List[int]]:
        level_to_node_values = defaultdict(list)

        def traverse(node: TreeNode, level: int) -> defaultdict[int, list[int]]:
            if node is None:
                return 
            level_to_node_values[level].append(node.val)
            traverse(node.left, level + 1)
            traverse(node.right, level + 1)

        traverse(root, 0)
        return [level_to_node_values[level] for level in sorted(level_to_node_values)]
```
