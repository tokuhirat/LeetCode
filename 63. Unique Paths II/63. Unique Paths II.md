# 63. Unique Paths II
## STEP1
- 何も見ずに解いてみる
- はじめはスタート地点に障害物が置かれているケースを考慮していなかった。
```python
class Solution:
    def uniquePathsWithObstacles(self, obstacleGrid: List[List[int]]) -> int:
        num_rows = len(obstacleGrid)
        num_columns = len(obstacleGrid[0])
        num_paths = [[0] * num_columns for _ in range(num_rows)]
        if obstacleGrid[0][0] == 1:
            return 0

        num_paths[0][0] = 1
        for row in range(num_rows):
            for column in range(num_columns):
                if row + 1 < num_rows and obstacleGrid[row + 1][column] == 0:
                    num_paths[row + 1][column] += num_paths[row][column]
                if column + 1 < num_columns and obstacleGrid[row][column + 1] == 0:
                    num_paths[row][column + 1] += num_paths[row][column]
        return num_paths[-1][-1]
```
#### memo
時間計算量: $O(m \times n)$  
空間計算量: $O(m \times n)$

```python
class Solution:
    def uniquePathsWithObstacles(self, obstacleGrid: List[List[int]]) -> int:
        num_rows = len(obstacleGrid)
        num_columns = len(obstacleGrid[0])
        num_paths = [[0] * num_columns for _ in range(num_rows)]
        if obstacleGrid[0][0] == 1:
            return 0

        num_paths[0][0] = 1
        for row in range(num_rows):
            for column in range(num_columns):
                if obstacleGrid[row][column] == 1:
                    continue
                if row != 0:
                    num_paths[row][column] += num_paths[row - 1][column]
                if column != 0:
                    num_paths[row][column] += num_paths[row][column - 1]
        return num_paths[-1][-1]
```

```python
class Solution:
    def uniquePathsWithObstacles(self, obstacleGrid: List[List[int]]) -> int:
        OBSTACLE = 1
        if obstacleGrid[0][0] == OBSTACLE:
            return 0
        num_rows = len(obstacleGrid)
        num_columns = len(obstacleGrid[0])
        num_paths = [[0] * num_columns for _ in range(num_rows)]
        num_paths[0][0] = 1
        for row in range(num_rows):
            for column in range(num_columns):
                if obstacleGrid[row][column] == OBSTACLE:
                    continue
                if row != 0:
                    num_paths[row][column] += num_paths[row - 1][column]
                if column != 0:
                    num_paths[row][column] += num_paths[row][column - 1]
        return num_paths[-1][-1]
```
