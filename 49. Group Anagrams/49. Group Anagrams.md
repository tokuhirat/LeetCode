# 49. Group Anagrams
## STEP1
- 何も見ずに解いてみる  
- 文字はlowercase English letters なので 26 種類。配列で個数カウントし、アナグラム判定もできる。key を letter, value を 個数として管理することもできる。こちらの方が入力文字が増えても対応しやすい。
- 実装したが動かないコード。dict は hashable でないので、dict の key にはできない。
```python
from collections import defaultdict


class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        letter_count_to_anagrams = defaultdict(list)
        for word in strs:
            letter_count = defaultdict(int)
            for letter in word:
                letter_count[letter] += 1
            letter_count_to_anagrams[letter_count].append(word)
        return letter_count_to_anagrams.values()

```

- 修正版。長さ 26 の list で管理し、tuple に変換することにした。
```python
from collections import defaultdict


class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        def get_index(letter):
            index = ord(letter) - ord("a")
            if not　(0　<= index < 26):
                raise ValueError(f"Input is not lowercase English letter: {letter}")
            return index

        letter_count_to_anagrams = defaultdict(list)
        for word in strs:
            letter_count = [0 for _ in range(26)]
            for letter in word:
                letter_count[get_index(letter)] += 1
            letter_count_to_anagrams[tuple(letter_count)].append(word)
        return list(letter_count_to_anagrams.values())
```
#### memo
n=len(strs), l=len(word) として  
時間計算量: $O(nl)$  
空間計算量: $O(nl)$

from collections import defaultdict


class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        def get_index(letter: str) -> int:
            index = ord(letter) - ord("a")
            if not (0 <= index < len(string.ascii_lowercase)):
                raise ValueError(f"Input: {letter} is not lowercase English letter.")
            return index

        anagram_group_to_words = defaultdict(list)
        for word in strs:
            letter_count = [0] * len(string.ascii_lowercase)
            for letter in word:
                letter_count[get_index(letter)] += 1
            anagram_group_to_words[tuple(letter_count)].append(word)
        return list(anagram_group_to_words.values())
