```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def detectCycle(self, head: Optional[ListNode]) -> Optional[ListNode]:
        seen_nodes = set()
        while head is not None:
            if head in seen_nodes:
                return head
            seen_nodes.add(head)
            head = head.next
        return None
        
```

```python
class Solution:
    def detectCycle(self, head: Optional[ListNode]) -> Optional[ListNode]:
        slow = head
        fast = head
        while True:
            if fast is None or fast.next is None:
                return None
            slow = slow.next
            fast = fast.next.next
            if fast is slow:
                break

        fast = head
        while fast is not slow:
            slow = slow.next
            fast = fast.next
        return fast
```
