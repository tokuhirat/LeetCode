# 53. Maximum Subarray
## STEP1
- 何も見ずに解いてみる
- 累積和を考えた時に、小さい index 側にある最小値と大きい index 側にいる最大値との差の最大値を求めればよい。
```python
class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        if not nums:
            raise ValueError("nums is empty")

        cumsum = [0]
        for num in nums:
            cumsum.append(cumsum[-1] + num)
        min_in_left = []
        for sum_ in cumsum:
            if not min_in_left:
                min_in_left.append(sum_)
                continue
            min_in_left.append(min(min_in_left[-1], sum_))
        max_in_right = []
        for sum_ in cumsum[::-1]:
            if not max_in_right:
                max_in_right.append(sum_)
                continue
            max_in_right.append(max(max_in_right[-1], sum_))
        max_in_right.reverse()
        largest_sum = max(
            max_in_right[i] - min_in_left[i - 1] for i in range(1, len(cumsum))
        )
        return largest_sum
```
#### memo
時間計算量: $O(n)$  
空間計算量: $O(n)$

    - 累積和を計算する過程でこれまでの最小値との差を取ればシンプルになる。
```python
class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        prefix_sum = 0
        min_prefix_sum = 0
        max_sum = float("-inf")
        for num in nums:
            prefix_sum += num
            max_sum = max(max_sum, prefix_sum - min_prefix_sum)
            min_prefix_sum = min(min_prefix_sum, prefix_sum)
        return max_sum
```

```python
class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        max_subarray_ending_here = [0] * len(nums)
        max_subarray_ending_here[0] = nums[0]
        for i, num in enumerate(nums[1:], start=1):
            max_subarray_ending_here[i] = max(
                num, max_subarray_ending_here[i - 1] + num
            )
        return max(max_subarray_ending_here)
```

```python
class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        subarray_sum = nums[0]
        max_subarray_sum = nums[0]
        for num in nums[1:]:
            subarray_sum = max(num, subarray_sum + num)
            max_subarray_sum = max(max_subarray_sum, subarray_sum)
        return max_subarray_sum
```
