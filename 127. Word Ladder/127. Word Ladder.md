# 127. Word Ladder
## STEP1
- 何も見ずに解いてみる
- まず変換できる一文字違いの単語を繋いだグラフを作り、最短経路を見つけるので BFS が適していると考えた。
- _differs_by_single_letter に枝狩りや、バリデーションの部分を削ると、たまにLeetCodeの実行時間に間に合うが本質ではない。

```python
from collections import defaultdict, deque


class Solution:
    def ladderLength(self, beginWord: str, endWord: str, wordList: List[str]) -> int:
        # if len(beginWord) != len(endWord):
        #     raise ValueError("length is not same")
        self.word_length = len(beginWord)

        word_graph = defaultdict(list)
        for i in range(len(wordList) - 1):
            for j in range(i + 1, len(wordList)):
                if not self._differs_by_single_letter(wordList[i], wordList[j]):
                    continue
                word_graph[wordList[i]].append(wordList[j])
                word_graph[wordList[j]].append(wordList[i])

        seen_words = set()
        word_steps_queue = deque()
        for word in wordList:
            if not self._differs_by_single_letter(beginWord, word):
                continue
            seen_words.add(word)
            word_steps_queue.append((word, 2))
        
        while word_steps_queue:
            word, steps = word_steps_queue.popleft()
            if word == endWord:
                return steps
            for next_word in word_graph[word]:
                if next_word in seen_words:
                    continue
                seen_words.add(next_word)
                word_steps_queue.append((next_word, steps + 1))
        return 0

    def _differs_by_single_letter(self, word1: str, word2: str) -> bool:
        # if self.word_length != len(word1) or self.word_length != len(word2):
        #     raise ValueError("length is not same.")

        different_letter_count = 0
        for c1, c2 in zip(word1, word2):
            if c1 != c2:
                different_letter_count += 1
            if different_letter_count > 1:
                return False
        return different_letter_count == 1
```
#### memo
S = len(beginWord), N = len(wordList) として  
時間計算量: $O(N^2S)$  
空間計算量: $O(N^2S)$

from collections import defaultdict, deque


class Solution:
    def _differs_by_single_letter(self, word1: str, word2: str) -> bool:
        # if self.word_length != len(word1) or self.word_length != len(word2):
        #     raise ValueError("length is not same.")
        different_letter_count = 0
        for c1, c2 in zip(word1, word2):
            if c1 != c2:
                different_letter_count += 1
            if different_letter_count > 1:
                return False
        return different_letter_count == 1
    
    def _construct_word_graph(self, beginWord: str, wordList: List[str]) -> defaultdict[str, list[str]]:
        word_graph = defaultdict(list)
        for i in range(len(wordList) - 1):
            for j in range(i + 1, len(wordList)):
                if not self._differs_by_single_letter(wordList[i], wordList[j]):
                    continue
                word_graph[wordList[i]].append(wordList[j])
                word_graph[wordList[j]].append(wordList[i])

        if beginWord not in wordList:
            for word in wordList:
                if not self._differs_by_single_letter(beginWord, word):
                    continue
                word_graph[word].append(beginWord)
                word_graph[beginWord].append(word)
        return word_graph

    def ladderLength(self, beginWord: str, endWord: str, wordList: List[str]) -> int:
        # if len(beginWord) != len(endWord):
        #     raise ValueError("length is not same")
        self.word_length = len(beginWord)
        word_graph = self._construct_word_graph(beginWord, wordList)

        seen_words = set([beginWord])
        word_steps_queue = deque([(beginWord, 1)])        
        while word_steps_queue:
            word, steps = word_steps_queue.popleft()
            if word == endWord:
                return steps
            for next_word in word_graph[word]:
                if next_word in seen_words:
                    continue
                seen_words.add(next_word)
                word_steps_queue.append((next_word, steps + 1))
        return 0

                pattern_to_words[pattern].append(word)

ジェネレータを使ったパターン
```python
from collections import defaultdict, deque
from collections.abc import Generator


class Solution:
    def ladderLength(self, beginWord: str, endWord: str, wordList: List[str]) -> int:
        def make_patterns(word: str) -> Generator[tuple[str, str], None, None]:
            for i in range(len(word)):
                yield (word[:i], word[i + 1:])

        pattern_to_words = defaultdict(list)
        for word in wordList:
            for pattern in make_patterns(word):
                pattern_to_words[pattern].append(word)

        def get_neighbors(word: str) -> Generator[str, None, None]:
            for pattern in make_patterns(word):
                for next_word in pattern_to_words[pattern]:
                    yield next_word

        queue = deque([(beginWord, 1)])  # (word, len_sequence)
        used_word = set([beginWord])
        while queue:
            word, len_sequence = queue.popleft()
            for next_word in get_neighbors(word):
                if next_word == endWord:
                    return len_sequence + 1
                if next_word in used_word:
                    continue
                used_word.add(next_word)
                queue.append((next_word, len_sequence + 1))
        return 0
```
