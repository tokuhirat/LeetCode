# 929. Unique Email Addresses
## STEP1
- 何も見ずに解いてみる
```python
class Solution:
    def numUniqueEmails(self, emails: List[str]) -> int:
        def get_actual_address(email):
            local_name, domain_name = email.split("@")
            actual_local_name = get_actual_local_name(local_name)
            return (actual_local_name, domain_name)
        
        def get_actual_local_name(local_name):
            actual_local_name = local_name.split("+")[0]  # ignore letters after "+" sign
            actual_local_name = actual_local_name.replace(".", "")  # remove "."
            return actual_local_name
        
        unique_emails = set()
        for email in emails:
            actual_address = get_actual_address(email)
            unique_emails.add(actual_address)
        return len(unique_emails)
```
#### memo
- local name と domain name に分割して処理し、ユニークな tuple の数を数える方針とした。
- 関数名は微妙な気がする。正規化みたいな名前の方が良いかもしれない。
- エラー処理はしていない。勘違いで local_name, domain_name = email.split("@") で@が複数ある場合 domain_name は list になりそうで複雑な記述が必要になると思っていた。後で試すと、@の個数が1でない場合、そのままで unpack に関する ValueError が出るので楽に処理できる。ちなみに x, *y = hoge.split("@") と書くと@の個数にかかわらずにエラーが起きず y が list になり、長さが変わる。

n = len(emails), m = email の長さとして  
時間計算量: $O(nm)$  
空間計算量: $O(nm)$  
制約では email の長さは最大 100

課題にある制約をチェックする。不正な場合はスキップする方針とした。
```python
class Solution:
    def numUniqueEmails(self, emails: List[str]) -> int:
        unique_emails = set()
        for email in emails:
            canonical_email = self._canonicalize_email(email)
            if canonical_email is not None:
                unique_emails.add(canonical_email)
        return len(unique_emails)
    
    def _canonicalize_email(self, email: str) -> Optional[str]:
        if not self._is_valid_email_format(email):
            return None
        local_name, domain_name = email.split("@")
        if not self._is_valid_local_name(local_name):
            return None
        if not self._is_valid_domain_name(domain_name):
            return None
        canonical_local_name = self._canonicalize_local_name(local_name)
        return f"{canonical_local_name}@{domain_name}"

    def _is_valid_email_format(self, email: str) -> bool:
        if len(email) > 100:
            return False
        if not self._is_valid_letters(email):
            return False
        if email.count("@") != 1:
            return False
        return True

    def _is_valid_letters(self, email: str) -> bool:
        return all(
            ord("a") <= ord(c) <= ord("z") or c in ".+@"
            for c in email
        )
    
    def _is_valid_local_name(self, local_name: str) -> bool:
        if not local_name:
            return False
        if local_name[0] == "+":
            return False
        return True
    
    def _is_valid_domain_name(self, domain_name: str) -> bool:
        if not domain_name:
            return False
        if domain_name[0] == ".":
            return False
        return True
    
    def _canonicalize_local_name(self, local_name) -> str:
        canonical_local_name = local_name.split("+")[0]
        canonical_local_name = canonical_local_name.replace(".", "")
        return canonical_local_name
```
