# 617. Merge Two Binary Trees
## STEP1
- 何も見ずに解いてみる
- 新たにTreeNodeをすべて作ることにする。
- None となりうる node の子を参照する方法が見やすく書けなかった。関数にすると見通しは良くなるかもしれない。
```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def mergeTrees(self, root1: Optional[TreeNode], root2: Optional[TreeNode]) -> Optional[TreeNode]:
        if root1 is None and root2 is None:
            return None
        
        root1_val = root1.val if root1 is not None else 0
        root2_val = root2.val if root2 is not None else 0
        root1_left = root1.left if root1 is not None else None
        root1_right = root1.right if root1 is not None else None
        root2_left = root2.left if root2 is not None else None
        root2_right = root2.right if root2 is not None else None
        
        merged_root = TreeNode(root1_val + root2_val)
        left_node = self.mergeTrees(root1_left, root2_left)
        right_node = self.mergeTrees(root1_right, root2_right)
        if left_node is not None:
            merged_root.left = left_node
        if right_node is not None:
            merged_root.right = right_node
        return merged_root
```
#### memo
n = 合計ノード数として  
時間計算量: $O(n)$  
空間計算量: $O(n)$

```python
class Solution:
    def mergeTrees(self, root1: Optional[TreeNode], root2: Optional[TreeNode]) -> Optional[TreeNode]:
        roots = list(filter(None, [root1, root2]))
        if not roots:
            return None
        root = TreeNode()
        stack = [(root, roots)]
        while stack:
            dst_node, src_nodes = stack.pop() 
            dst_node.val = sum(n.val for n in src_nodes)
            src_lefts = [n.left for n in src_nodes if n.left is not None]
            if src_lefts:
                dst_node.left = TreeNode()
                stack.append((dst_node.left, src_lefts))
            src_rights = [n.right for n in src_nodes if n.right is not None]
            if src_rights:
                dst_node.right = TreeNode()
                stack.append((dst_node.right, src_rights))
        return root
```

```python
class Solution:
    def mergeTrees(self, root1: Optional[TreeNode], root2: Optional[TreeNode]) -> Optional[TreeNode]:
        if root1 is None and root2 is None:
            return None
        
        merged_root = TreeNode()
        root1_left = None
        root1_right = None
        root2_left = None
        root2_right = None
        if root1 is not None:
            merged_root.val += root1.val
            root1_left = root1.left
            root1_right = root1.right
        if root2 is not None:
            merged_root.val += root2.val
            root2_left = root2.left
            root2_right = root2.right

        merged_root.left = self.mergeTrees(root1_left, root2_left)
        merged_root.right = self.mergeTrees(root1_right, root2_right)
        return merged_root
```
