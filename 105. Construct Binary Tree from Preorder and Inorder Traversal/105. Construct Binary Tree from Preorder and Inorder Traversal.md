```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def buildTree(self, preorder: List[int], inorder: List[int]) -> Optional[TreeNode]:
        node_val_to_inorder_index = {val: index for index, val in enumerate(inorder)}

        preorder_index = 0
        def build_tree_from_inorder_index(low: int, high: int) -> Optional[TreeNode]:
            if low >= high:
                return None
            nonlocal preorder_index
            inorder_split_index = node_val_to_inorder_index[preorder[preorder_index]]
            root = TreeNode(preorder[preorder_index])
            preorder_index += 1
            root.left = build_tree_from_inorder_index(low, inorder_split_index)
            root.right = build_tree_from_inorder_index(inorder_split_index + 1, high)
            return root

        return build_tree_from_inorder_index(0, len(preorder))
```

```python
class Solution:

        if not preorder or not inorder:
            return None
        
        root = TreeNode(preorder[0])
        num_left_tree_nodes = inorder.index(root.val)
        root.left = self.buildTree(
            preorder[1 : num_left_tree_nodes + 1], inorder[:num_left_tree_nodes]
        )
        root.right = self.buildTree(
            preorder[num_left_tree_nodes + 1 :], inorder[num_left_tree_nodes + 1 :]
```

```python
from array import array


class Solution:
    def buildTree(self, preorder: List[int], inorder: List[int]) -> Optional[TreeNode]:
        preorder = array('l', preorder)
        inorder = array('l', inorder)
        self.val_to_inorder_index = {val: index for index, val in enumerate(inorder)}
        return self._buildTree(memoryview(preorder), memoryview(inorder))

    def _buildTree(self, preorder, inorder):
        if len(preorder) == 0 or len(inorder) == 0:
            return None
        root = TreeNode(preorder[0])
        inorder_start_index = self.val_to_inorder_index[inorder[0]]
        inorder_split_index = self.val_to_inorder_index[root.val]
        num_left_tree_nodes = inorder_split_index - inorder_start_index

        root.left = self._buildTree(
            preorder[1 : num_left_tree_nodes + 1], inorder[:num_left_tree_nodes]
        )
        root.right = self._buildTree(
            preorder[num_left_tree_nodes + 1 :], inorder[num_left_tree_nodes + 1 :]
        )
        return root
```

```python
from collections import namedtuple


class Solution:
    def buildTree(self, preorder: List[int], inorder: List[int]) -> Optional[TreeNode]:
        if not preorder:
            return None
        
        IndexRange = namedtuple("IndexRange", ["start", "stop"])

        val_to_inorder_inde = {val: index for index, val in enumerate(inorder)}
        root = TreeNode()
        stack = [(root, IndexRange(0, len(preorder)), IndexRange(0, len(inorder)))]
        # (node, (preorder_range), (inorder_range))
        while stack:
            node, preorder_range, inorder_range = stack.pop()
            node.val = preorder[preorder_range.start]
            inorder_split_index = val_to_inorder_inde[node.val]
            left_counts = inorder_split_index - inorder_range.start
            right_counts = inorder_range.stop - (inorder_split_index + 1)
            if left_counts > 0:
                left_node = TreeNode()
                node.left = left_node
                left_preorder_range = IndexRange(
                    preorder_range.start + 1, preorder_range.start + left_counts + 1,
                )
                left_inorder_range = IndexRange(inorder_range.start, inorder_split_index)
                stack.append((left_node, left_preorder_range, left_inorder_range))
            if right_counts > 0:
                right_node = TreeNode()
                node.right = right_node
                right_preorder_range = IndexRange(
                    preorder_range.start + left_counts + 1, preorder_range.stop,
                )
                right_inorder_range = IndexRange(inorder_split_index + 1, inorder_range.stop)
                stack.append((right_node, right_preorder_range, right_inorder_range))
        return root
```
