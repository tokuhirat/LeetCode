# 98. Validate Binary Search Tree
## STEP1
- 何も見ずに解いてみる
- 部分木に含まれる値の最大値や最小値も管理しないといけない。DFS の方が素直と感じる。
- 再帰で書く場合、返り値をどうしたらよいか。部分木が BST であるか、含まれる最大最小の値を返すと良さそう。かなり複雑になった。
- 通りがけで見ていくと値が昇順になることを利用するパターンもありそう。書き方が思いつかなかった。（振り返ると手作業でやるステップをサボったからな気がする。inorder (stack) は手作業でやってみたら書けそう）

```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        def check_BST_get_min_max(root: TreeNode) -> tuple[bool, int, int]:
            if root.left is None and root.right is None:
                return (True, root.val, root.val)

            if root.left is not None:
                is_left_BST, min_val_in_left, max_val_in_left = check_BST_get_min_max(root.left)
                if not is_left_BST:
                    return (False, 0, 0)
                if max_val_in_left >= root.val:
                    return (False, 0, 0)
            else:
                min_val_in_left = root.val

            if root.right is not None:
                is_right_BST, min_val_in_right, max_val_in_right = check_BST_get_min_max(root.right)
                if not is_right_BST:
                    return (False, 0, 0)
                if root.val >= min_val_in_right:
                    return (False, 0, 0)
            else:
                max_val_in_right = root.val

            return (True, min_val_in_left, max_val_in_right)
        
        return check_BST_get_min_max(root)[0]
```
- (False, 0, 0) の 0 は参照されることがないので (False, None, None) の方がよいかもしれない。

空間計算量: $O(n)$

```python
import math


class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        def is_valid_BST_node(node: Optional[TreeNode], low: int, high: int) -> bool:
            if node is None:
                return True
            if not (low < node.val < high):
                return False
            return (
                is_valid_BST_node(node.left, low, node.val) and
                is_valid_BST_node(node.right, node.val, high)
            )
        
        return is_valid_BST_node(root, -math.inf, math.inf)
```

preorder (stack)
```python
import math


class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        if root is None:
            return True

        stack = [(root, -math.inf, math.inf)]
        while stack:
            node, low, high = stack.pop()
            if not (low < node.val < high):
                return False
            if node.left is not None:
                stack.append((node.left, low, node.val))
            if node.right is not None:
                stack.append((node.right, node.val, high))
        return True
```
inorder (再帰)
```python
import math


class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        min_value = -math.inf
        
        def isValidBSTHelper(root: Optional[TreeNode]) -> bool:
            nonlocal min_value
            if root is None:
                return True
            
            if not isValidBSTHelper(root.left):
                return False
            if root.val <= min_value:
                return False
            min_value = root.val
            if not isValidBSTHelper(root.right):
                return False
            return True

        return isValidBSTHelper(root)
```

inorder_sort に yield を使う手法
```python
import math


class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:        
        def inorder_sort(root: Optional[TreeNode]):
            if root is None:
                return
            if root.left is not None:
                yield from inorder_sort(root.left)
            yield root
            if root.right is not None:
                yield from inorder_sort(root.right)
        
        prev_value = -math.inf
        for node in inorder_sort(root):
            if node.val <= prev_value:
                return False
            prev_value = node.val
        return True
```

inorder (stack)
```python
import math


class Solution:
    def isValidBST(self, root: Optional[TreeNode]) -> bool:
        prev_value = -math.inf
        nodes = []

        def append_node_and_all_left_nodes(node: Optional[TreeNode]):
            while node:
                nodes.append(node)
                node = node.left
        
        append_node_and_all_left_nodes(root)
        while nodes:
            node = nodes.pop()
            if node.val <= prev_value:
                return False
            prev_value = node.val
            append_node_and_all_left_nodes(node.right)
        return True
```
