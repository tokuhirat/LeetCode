# 373. Find K Pairs with Smallest Sums
## STEP1
- 何も見ずに解いてみる
```python
import heapq


class Solution:
    def kSmallestPairs(self, nums1: List[int], nums2: List[int], k: int) -> List[List[int]]:
        k_smallest_sum = []  # min heap. sum * (-1) is stored.
        for n1 in nums1:
            for n2 in nums2:
                negative_sum = -(n1 + n2)
                if len(k_smallest_sum) < k:
                    heapq.heappush(k_smallest_sum, (negative_sum, [n1, n2]))
                    continue
                if negative_sum <= k_smallest_sum[0][0]:  # cannot find small sum
                    break
                # find small sum
                heapq.heappush(k_smallest_sum, (negative_sum, [n1, n2]))
                if len(k_smallest_sum) > k:
                    heapq.heappop(k_smallest_sum)
        return [n1_n2 for (_, n1_n2) in k_smallest_sum]

```
#### memo
- はじめ、nums1とnums2のindexをそれぞれ1ずつ大きくしていく方法を考えたが、戻るケースがあり誤っていることに気がついた。
- 手作業やる場合を考えると、nums1とnums2が縦横に書いた表の左上を埋めていくと考えて、二重ループで最小値を更新しない段階でbreakすると良いのではと実装した。計算量は自信がない。

時間計算量: $O(len(nums1) + k\log k)$ （自信なし）  
空間計算量: $O(k)$

```python
import heapq


class Solution:
    def kSmallestPairs(self, nums1: List[int], nums2: List[int], k: int) -> List[List[int]]:
        # min heap containing (sum, [index1, index2])
        candidates = [(nums1[0] + nums2[0], [0, 0])]
        k_smallest_pairs = []
        index_pair_already_added = set()

        def heappush_if_necessary(i, j):
            if not (0 <= i < len(nums1) and 0 <= j < len(nums2)):
                return
            if (i == 0 or j == 0 or 
                ((i - 1, j) in index_pair_already_added and
                (i, j - 1) in index_pair_already_added)
            ):
                heapq.heappush(candidates, (nums1[i] + nums2[j], [i, j]))

        while candidates and len(k_smallest_pairs) < k:
            _, (index1, index2) = heapq.heappop(candidates)
            k_smallest_pairs.append([nums1[index1], nums2[index2]])
            index_pair_already_added.add((index1, index2))
            heappush_if_necessary(index1 + 1, index2)
            heappush_if_necessary(index1, index2 + 1)
        return k_smallest_pairs
```
時間計算量: $O(k\log k)$  
空間計算量: $O(k)$

```python
import heapq


class Solution:
    def kSmallestPairs(self, nums1: List[int], nums2: List[int], k: int) -> List[List[int]]:
        # min heap containing (sum, [index1, index2]) 
        candidates = [(nums1[0] + nums2[0], [0, 0])]
        k_smallest_pairs = []
        index_pairs_already_added = set()

        def need_to_add(i, j):
            if i >= len(nums1) or j >= len(nums2):
                return False
            if i == 0 or j == 0:
                return True
            return (i - 1, j) in index_pairs_already_added and (i, j - 1) in index_pairs_already_added
        
        def heappush_if_necessary(i, j):
            if need_to_add(i, j):
                heapq.heappush(candidates, (nums1[i] + nums2[j], [i, j]))
        
        while candidates and len(k_smallest_pairs) < k:
            _, (index1, index2) = heapq.heappop(candidates)
            index_pairs_already_added.add((index1, index2))
            k_smallest_pairs.append([nums1[index1], nums2[index2]])
            heappush_if_necessary(index1 + 1, index2)
            heappush_if_necessary(index1, index2 + 1)
        return k_smallest_pairs
```
